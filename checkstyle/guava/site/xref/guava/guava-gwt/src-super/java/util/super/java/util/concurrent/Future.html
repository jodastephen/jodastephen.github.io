17  package java.util.concurrent;
18  
19  /**
20   * Emulation of Future. Since GWT environment is single threaded, attempting to block on the future
21   * by calling {@link #get()} or {@link #get(long, TimeUnit)} when the future is not yet done is
22   * considered illegal because it would lead to a deadlock. Future implementations must throw {@link
23   * IllegalStateException} to avoid a deadlock.
24   *
25   * @param <V> value type returned by the future.
26   */
27  public interface Future<V> {
28     boolean cancel(boolean mayInterruptIfRunning);
29  
30     boolean isCancelled();
31  
32     boolean isDone();
33  
34    // Even though the 'get' methods below are blocking, they are the only built-in APIs to get the
35    // result of the {@code Future}, hence they are not removed. The implementation must throw {@link
36    // IllegalStateException} if the {@code Future} is not done yet (see the class javadoc).
37  
38     V get() throws InterruptedException, ExecutionException;
39  
40     V get(long timeout, TimeUnit unit)
41         throws InterruptedException, ExecutionException, TimeoutException;
42  }
